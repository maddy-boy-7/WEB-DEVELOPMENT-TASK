  <script>
    // JavaScript for scroll effect
    const navbar = document.getElementById('navbar');

    window.addEventListener('scroll', () => {
      if (window.scrollY > 50 && window.scrollY <= 150) {
        navbar.className = 'color-red';
      } else if (window.scrollY > 150 && window.scrollY <= 300) {
        navbar.className = 'color-blue';
      } else if (window.scrollY > 300) {
        navbar.className = 'color-green';
      } else {
        navbar.className = '';
      }
    });
  </script>
